#전국 추가 확진 그래프 *년도별 *월별         
    def cov_year_month_all(self) :
        
        import numpy as np
        import pandas as pd
        import seaborn as sns
        import matplotlib.pyplot as plt
        import seaborn as sns
        import scipy.stats as stats
        
        self.final = self.df.pivot_table(index = ["년도별","월별"], 
                                         values = "전국 추가 확진", aggfunc=["mean", "sum"]).reset_index()
        
        
        self.w = 0.5
        self.idxs = np.array([0, 1, 2, 3, 4, 5, 6])


        plt.rc("font" , family = "Malgun Gothic")

        self.x1 = self.final["월별"].unique()
        self.x1 = sorted(self.x1)
        self.x1= self.x1[3:10]
        self.y1 = self.final[self.final["년도별"] == 2020]["mean"]
        self.y2 = self.final[self.final["년도별"] == 2021]["mean"]
        self.y3 = self.final[self.final["년도별"] == 2022]["mean"]

        self.y1 = self.y1.loc[2:8 ]["전국 추가 확진"]
        self.y2 = self.y2.loc[14:20 ]["전국 추가 확진"]
        self.y3=self.y3.loc[26: ]["전국 추가 확진"]

        self.x1=np.array(self.x1)

        plt.bar(self.idxs-self.w/4, np.log(self.y1), width = self.w/4, label = 2020)
        plt.bar(self.idxs, np.log(self.y2), width = self.w/4, label = 2021)
        plt.bar(self.idxs+self.w/4, np.log(self.y3), width = self.w/4, label = 2022)
        plt.xticks(self.idxs, self.x1)
        plt.legend(loc = "best")
        plt.show()
        
#월별 추가 확진

    def month_cov(self,year) :
        
        import numpy as np
        import pandas as pd
        import matplotlib.pyplot as plt
        import seaborn as sns
        import scipy.stats as stats

        self.df_year = self.df[self.df["년도별"] == year]
        self.df_month = self.df_year.pivot_table(index = "월별" ,
                                           values = ["서울시 추가 확진", "전국 추가 확진"], aggfunc = "mean")
        
        plt.title(f'{year}년의 월별 추가 확진자 수')
        plt.plot(self.df_month["서울시 추가 확진"], label = "서울시 추가 확진")
        plt.plot(self.df_month["전국 추가 확진"], label = "전국 추가 확진")
        plt.xlabel("월별")
        plt.ylabel("추가 확진자 수")
        plt.legend(loc = "best")
        plt.show()

    def quarter_years(self , year) :

        import pandas as pd
        import matplotlib.pyplot as plt
        import seaborn as sns

        plt.title(f'{year}년도 분기별 추가 확진자 동향')
        plt.bar(self.x, self.y, color='gray')
        plt.xticks(self.x)
        plt.show()

# 특정년도 추가확진자 수 평균검증   
    def test_cov_year_month_all(self,year) :
        
        import numpy as np
        import pandas as pd
        import matplotlib.pyplot as plt
        import seaborn as sns
        import scipy.stats as stats
        
        self.df_year = self.df.pivot_table(index = "년도별", 
                                         values = "전국 추가 확진", aggfunc="mean")
        
        self.test_df_year = self.df_year.loc[year, "전국 추가 확진"]
        
        self.df_test_year = stats.ttest_1samp(self.df_year["전국 추가 확진"], self.test_df_year)

        if self.df_test_year[1] > 0.05 :
            print(f'귀무가설 채택, 대립가설 기각 : {year}년도의 추가확진자 수와 추가확진자 수의 평균의 차이가 없다')
            
        else :
            print(f'귀무가설 기각, 대립가설 채택 : {year}년도의 추가확진자 수와 추가확진자 수의 평균의 차이가 있다')
