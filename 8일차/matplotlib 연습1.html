import matplotlib.pyplot as plt
import numpy as np

data1 = np.arange(1, 20)
data2 = data1 * 3
data3 = data1 * 3 + 5
data4 = data1 * 3 - 5
 
plt.rc("font", family="Malgun Gothic")   # 한글이 깨지지 않게 해준다
plt.plot(data1, data2, "ro-", data1, data3, "gd", data1, data4, "m^:")
plt.xlabel("x 축")
plt.ylabel("y 축")
plt.show()
